    ```javascript
    // ─── mergeVerdicts (D-03 merge rule) ───────────────────────────────────
    // Inputs:
    //   detFindings: array of deterministic findings (from runDeterministicScreen
    //                additive ban-list hits — may be empty).
    //   llmVerdict:  full verdict object from LLM pass (or null if no LLM ran).
    //
    // Output: merged verdict object conforming to the schema validateVerdict enforces.
    //
    // Rules:
    //   - combinedFindings = detFindings concat llmVerdict.findings
    //   - severity = max severity in combinedFindings (blocking > material > nice-to-have)
    //   - decision:
    //       - if any blocking in combinedFindings:
    //           - if ANY blocking finding points at OBJECTIVES.md location → DRIFTED-objective-needs-update
    //           - else → DRIFTED-output-needs-revision
    //       - else: decision = ALIGNED
    //   - findings_count = combinedFindings.length
    //   - rationale = llmVerdict.rationale (fallback: synthesized from findings)
    function mergeVerdicts(detFindings, llmVerdict) {
      const llm = llmVerdict || { decision: 'ALIGNED', severity: 'nice-to-have', findings: [], rationale: 'no semantic findings' };
      const combinedFindings = [].concat(detFindings || [], llm.findings || []);
      // Severity max
      const SEV_ORDER = { blocking: 3, material: 2, 'nice-to-have': 1 };
      let maxSev = 'nice-to-have';
      for (const f of combinedFindings) {
        if ((SEV_ORDER[f.severity] || 0) > (SEV_ORDER[maxSev] || 0)) maxSev = f.severity;
      }
      // Decision derivation
      let decision;
      if (maxSev === 'blocking') {
        const blockingFinding = combinedFindings.find((f) => f.severity === 'blocking');
        const loc = blockingFinding ? blockingFinding.location || '' : '';
        if (/OBJECTIVES\.md/i.test(loc) || /frontmatter/i.test(loc)) {
          decision = 'DRIFTED-objective-needs-update';
        } else {
          decision = 'DRIFTED-output-needs-revision';
        }
      } else {
        decision = 'ALIGNED';
      }
      return {
        decision,
        severity: maxSev,
        findings_count: combinedFindings.length,
        findings: combinedFindings,
        rationale: llm.rationale || `merged verdict from ${combinedFindings.length} findings`,
      };
    }

    // ─── runAlign (entry point for workflow + tests) ───────────────────────
    // opts = {
    //   candidate: <path to candidate artifact>,
    //   baseline:  <path to baseline OBJECTIVES.md>,
    //   verdictOutPath: <path — defaults to .planning/.align-verdict.tmp.json>,
    //   llmPass: <function or null — injectable for tests; production: spawns Task>,
    // }
    //
    // Returns the merged verdict object (also writes it to verdictOutPath).
    // The llmPass parameter exists for testability — in production, the
    // workflow markdown (align-gate.md Step 2) invokes the subagent Task
    // directly and passes the workflow-level verdict back to `commit` via
    // the --verdict flag. This function is what drives the deterministic
    // short-circuit + LLM merge pipeline inside a single call surface.
    function runAlign(cwd, opts) {
      const candidate = opts.candidate;
      const baseline = opts.baseline;
      const verdictOutPath = opts.verdictOutPath || path.join(planningPaths(cwd).planning, '.align-verdict.tmp.json');
      const llmPass = opts.llmPass; // may be null/undefined
      // Step 1: deterministic screen
      const screen = runDeterministicScreen(cwd, { candidate, baseline });
      if (screen.verdict) {
        // Short-circuit: deterministic fired blocking or material — no LLM pass.
        writeVerdict(verdictOutPath, screen.verdict);
        return screen.verdict;
      }
      // Step 2: LLM pass (injected for tests; in production, the workflow
      // markdown drives the Task spawn — this function is not called from
      // the workflow for the LLM path since the workflow handles Task itself.
      // For callers that want a single-function pipeline (and tests), llmPass
      // is invoked here.)
      let llmVerdict = null;
      if (typeof llmPass === 'function') {
        llmVerdict = llmPass({
          candidate, baseline, verdictOutPath, cwd,
          alreadyKnownFindings: screen.findings,
        });
      }
      // Step 3: merge
      const merged = mergeVerdicts(screen.findings, llmVerdict);
      writeVerdict(verdictOutPath, merged);
      return merged;
    }
    ```
