    ```javascript
    //   verdict   — full verdict object (already validated)
    //   opts.korea — boolean (Korea signal detected → Korean headers/body)

    // Returns: full markdown content with --- frontmatter --- header + body
    function renderAlignReport(verdict, opts) {
      const korea = !!(opts && opts.korea);
      const override = !!(opts && opts.override);
      const overrideReason = (opts && opts.overrideReason) || '';
      const now = new Date().toISOString();
      const fm = {
        decision: override ? 'ALIGNED' : verdict.decision,
        severity: verdict.severity,
        findings_count: verdict.findings_count,
        at: now,
        ...(override ? { override: true, override_reason: overrideReason } : {}),
      };
      const fmYaml = reconstructFrontmatter(fm);
      // Body
      const findingsHeader = korea ? '## Findings / 발견사항' : '## Findings';
      const rationaleHeader = korea ? '## Rationale / 이유' : '## Rationale';
      const overrideHeader = korea ? '## User Override / 사용자 승인' : '## User Override';
      const findingsList = verdict.findings.map((f) => `- **[${f.severity}]** \`${f.location}\` — ${f.description}`).join('\n');
      const bodyParts = [
        findingsHeader,
        '',
        verdict.findings.length > 0 ? findingsList : (korea ? '_(추가 발견사항 없음)_' : '_(no additional findings)_'),
        '',
        rationaleHeader,
        '',
        verdict.rationale,
        '',
      ];

          '',

          '',

      const planningDir = path.resolve(cwd, '.planning');
      // startsWith check WITH trailing separator to prevent /.planning-other/ matches
    ```

    ```javascript
              if (screen.verdict) {
                // Short-circuit: write the verdict and exit signaling short_circuited.
                align.writeVerdict(verdictOutPath || path.join(core.planningPaths(cwd).planning, '.align-verdict.tmp.json'), screen.verdict);
                core.output({ short_circuited: true, verdict: screen.verdict }, raw, 'short_circuited');
              } else {
                core.output({ short_circuited: false, deterministic_findings: screen.findings }, raw, 'llm_pass_needed');
              }
            } catch (err) {
              // Route all errors through core.error so Node's default uncaught-
              // exception handler never prints the stack trace + absolute paths.
              // core.error writes "Error: <msg>" to stderr and process.exit(1)s.
              core.error(err.message);
            }
            break;
          }
          if (subcommand === 'commit') {
            const verdictPath = vIdx !== -1 ? args[vIdx + 1] : null;
            const overrideReason = reasonIdx !== -1 ? args[reasonIdx + 1] : null;
            if (!verdictPath) {
              core.error('align commit requires --verdict <path>');
              break;
            }
            // T-04-01 + Test 10: wrap the commit call so thrown errors (e.g.,
            // path traversal from _resolveSafePath) route through core.error,
            // which emits a sanitized stderr line and exits 1. Without this
            // try/catch, Node's default uncaught-exception handler prints the
            // full stack trace — including the absolute path to align.cjs —
            // leaking the developer cwd and failing the doesNotMatch assertion
            // `/\/Users\/[a-z0-9_-]+\/GSD-for-Business/i` in the CLI-layer test.
            try {
              const result = align.commitAlignVerdict(cwd, {
                verdictPath,
                override,
                overrideReason,
              });
              core.output(result, raw, `ALIGN-00.md written at ${result.alignPath}`);
            } catch (err) {
              core.error(err.message);
            }
            break;
          }
          core.error(`align: unknown subcommand '${subcommand}'. Valid: run, commit`);
          break;
        }
    ```

    NB: `core.error(msg)` calls `process.exit(1)` — it never returns. The trailing `break;` after each `core.error` is defensive and unreachable but harmless. The `try/catch` must wrap BOTH the `align.runDeterministicScreen`+`align.writeVerdict` pair (in the `run` branch) AND the `align.commitAlignVerdict` call (in the `commit` branch). Any thrown `Error` — path traversal, JSON parse failure, sanitization failure, state lock contention — MUST be caught here and forwarded through `core.error(err.message)`. Leaving any `throw` uncaught causes Node to print the full stack with the absolute path to `brief/bin/lib/align.cjs`, which Test 10 asserts MUST NOT appear in output.

No new `path = require('path')` import is needed since align.cjs composes it; we reference `path` via the `core` module's planningPaths helper.
