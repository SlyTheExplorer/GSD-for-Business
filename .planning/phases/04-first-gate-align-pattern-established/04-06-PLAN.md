    function extractStepBody(content, stepHeadingRegex) {
      const lines = content.split('\n');
      const startIdx = lines.findIndex((l) => stepHeadingRegex.test(l));
      if (startIdx === -1) return null;
      // Find the next `## Step ` heading AFTER startIdx
      let endIdx = lines.length;
      for (let i = startIdx + 1; i < lines.length; i++) {
        if (/^## Step /.test(lines[i])) { endIdx = i; break; }
      }
      return lines.slice(startIdx, endIdx).join('\n');

    const { test } = require('node:test');
    const assert = require('node:assert');
    const fs = require('node:fs');
    const path = require('node:path');

    const REPO_ROOT = path.join(__dirname, '..');

    function walkFiles(dir) {
      const out = [];
      if (!fs.existsSync(dir)) return out;
      for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
        const full = path.join(dir, entry.name);
        if (entry.isDirectory()) out.push(...walkFiles(full));
        else if (entry.isFile()) out.push(full);
      }
      return out;
    }

    test('hooks/ directory contains no file referencing align-gate (ROADMAP Phase 4 SC #3)', () => {
      const hooksDir = path.join(REPO_ROOT, 'hooks');
      if (!fs.existsSync(hooksDir)) {
        // hooks/ not present — vacuously passes
        return;
      }
      const hits = [];
      for (const file of walkFiles(hooksDir)) {
        let content;
        try {
          content = fs.readFileSync(file, 'utf-8');
        } catch {
          continue; // binary or unreadable — skip
        }
        for (const needle of ['align-gate', 'brief-align-gate', 'align.cjs']) {
          if (content.includes(needle)) {
            hits.push(`${file} contains '${needle}'`);
          }
        }
      }
      if (hits.length > 0) {
        assert.fail(
          `ALIGN gate must NOT be referenced in hooks/ (ROADMAP Phase 4 Success Criterion #3, RESEARCH Anti-pattern #2). Hits:\n  ${hits.join('\n  ')}`
        );
      }

      const content = fs.readFileSync(path.join(REPO_ROOT, 'agents/brief-align-gate.md'), 'utf-8');
      const fmMatch = content.match(/^---\n([\s\S]*?)\n---/);
      assert.ok(fmMatch, 'frontmatter must exist');
