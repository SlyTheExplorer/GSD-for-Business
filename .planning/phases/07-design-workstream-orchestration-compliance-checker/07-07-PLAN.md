       // Phase 7 — Recommended next workstream derivation (D-07 + D-08 derive-at-read).
       // Reads loadWorkstreams() and finds the first workstream where:
       //   - workstream itself NOT in state.brief.workstreams_completed
       //   - all depends_on are in workstreams_completed
       const { loadWorkstreams } = require('./workstream-loader.cjs');
       function computeRecommendedNext(cwd, briefState) {
         try {
           const ws = loadWorkstreams(cwd);
           const completed = new Set(Array.isArray(briefState && briefState.workstreams_completed) ? briefState.workstreams_completed : []);
           // Soft-order: BMC → GTM → BRAND → OPERATIONS → FINANCIAL → RISK → ROADMAP → TECH-ARCH → COMPLIANCE
           const SOFT_ORDER = [
             'business-model-canvas', 'go-to-market', 'brand', 'operations',
             'financial', 'risk', 'roadmap', 'tech-arch', 'compliance'
           ];
           const orderedWs = SOFT_ORDER
             .map(slug => ws.find(w => (w.slug || w.name) === slug))
             .filter(Boolean)
             .concat(ws.filter(w => !SOFT_ORDER.includes(w.slug || w.name)));
           for (const s of orderedWs) {
             const slug = s.slug || s.name;
             if (completed.has(slug)) continue;
             const depsOk = (s.depends_on || []).every(d => completed.has(d));
             if (depsOk) return slug;
           }
           return '—';
         } catch {
           return '—';
         }
       }
