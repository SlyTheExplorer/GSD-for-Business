function deriveAudienceDefaults(business_model, region) {

const { test } = require('node:test');
const assert = require('node:assert');
const fs = require('node:fs');
const path = require('node:path');
const os = require('node:os');

const { buildBusinessContext, COMPLIANCE_PACK_TO_REFERENCE } = require('../brief/bin/lib/context-inject.cjs');
const FIXTURES = path.join(__dirname, 'fixtures', 'context-inject');

function setupTmp(configSrc, objectivesSrc) {
  const tmp = fs.mkdtempSync(path.join(os.tmpdir(), 'brief-ctx-'));
  const planning = path.join(tmp, '.planning');
  fs.mkdirSync(planning, { recursive: true });
  if (configSrc) fs.copyFileSync(path.join(FIXTURES, configSrc), path.join(planning, 'config.json'));
  if (objectivesSrc) fs.copyFileSync(path.join(FIXTURES, objectivesSrc), path.join(planning, 'OBJECTIVES.md'));
  return tmp;
}

test('KR/B2B/fintech: language=ko, formal tone, required_reading includes PIPA+ISMS-P', () => {
  const cwd = setupTmp('config-kr-b2b.json', 'objectives-kr-b2b-fintech.md');
  const ctx = buildBusinessContext({ cwd });
  assert.equal(ctx.language, 'ko');
  assert.equal(ctx.korea_signal, true);
  assert.equal(ctx.business_model, 'b2b');
  assert.equal(ctx.region, 'kr');
  assert.deepEqual(ctx.compliance_packs, ['PIPA', 'ISMS-P']);
  assert.equal(ctx.audienceDefaults['voice.tone'], 'formal');
  assert.equal(ctx.audienceDefaults['voice.perspective'], 'first-person-plural');
  assert.ok(ctx.requiredReading.includes(COMPLIANCE_PACK_TO_REFERENCE['PIPA']));
  assert.ok(ctx.requiredReading.includes(COMPLIANCE_PACK_TO_REFERENCE['ISMS-P']));
  assert.match(ctx.promptBlock, /<business_context>/);
  assert.match(ctx.promptBlock, /<language>ko<\/language>/);
});

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `.planning/config.json` read | User-authored JSON file; may contain malformed data or PII |
| `.planning/OBJECTIVES.md` body read | User-authored markdown; passed to detectKoreaSignals for Hangul scan |
| `buildBusinessContext()` return → Task prompt | The `promptBlock` string is interpolated into LLM prompts — any injection risk surfaces downstream |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-5-01-01 | Tampering | `brief/bin/lib/context-inject.cjs` COMPLIANCE_PACK_TO_REFERENCE map | mitigate | `Object.freeze()` on the map prevents runtime mutation; map value paths are in `brief/references/compliance/korea/*` which Plan 06 ships verbatim |
| T-5-01-02 | Information Disclosure | config.json read — malformed JSON | mitigate | try/catch around JSON.parse; returns {} (empty brief namespace) rather than throwing, so downstream callers never see partial/malformed state |
| T-5-01-03 | Injection | OBJECTIVES.md body → detectKoreaSignals → language choice → promptBlock | accept | detectKoreaSignals is read-only regex; output is 'ko' or 'en' enum (closed set); no unescaped user text flows into promptBlock beyond business_model/region/language/audience_policy — all of which are compared against closed enums or enum-like fields |
| T-5-01-04 | Elevation of Privilege | requiredReading paths escape allowlist | mitigate | requiredReading entries come only from the frozen COMPLIANCE_PACK_TO_REFERENCE map — no user-supplied strings reach the path |
| T-5-01-05 | Denial of Service | Large OBJECTIVES.md body stalls detectKoreaSignals regex | accept | Existing Phase 3 detectKoreaSignals handles arbitrary body size; no known ReDoS in the Hangul + keyword regex; not a new attack surface |
