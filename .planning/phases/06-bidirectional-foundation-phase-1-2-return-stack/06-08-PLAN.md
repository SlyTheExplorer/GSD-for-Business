// Phase 6 authored files — excluding the vocabulary reference (which DOCUMENTS the ban-list as forbidden).

function banHitsOutsideBanList(content) {
  // Lines containing 'ban-list' / 'BAN_' / 'ban list' / '금지' are documenting the ban itself; exclude from audit.
  const hits = [];
  const lines = content.split('\n');
  for (let i = 0; i < lines.length; i++) {
    const l = lines[i];
    if (/ban[- ]?list|BAN_|금지/i.test(l)) continue;
    for (const re of [BAN_EN, BAN_KO, BAN_SYMBOL]) {
      re.lastIndex = 0;
      let m;
      while ((m = re.exec(l)) !== null) {
        hits.push({ token: m[0], line: i + 1 });
      }
    }
  }
  return hits;
}
