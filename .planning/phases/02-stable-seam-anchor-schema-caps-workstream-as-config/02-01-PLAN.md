    ```javascript
    /**
     * Serialize a JS value as YAML content lines (no leading key). Recursively handles:
     * - null/undefined: emits the literal string 'null'
     * - string/number/boolean: emits scalar (quoted if contains : or #)
     * - array of scalars: block list `- item` (one per line) or inline `[a, b, c]` for short all-string arrays
     * - array of objects: block list `- ` with indented map body
     * - object (map): indented key: value map; recurses for nested objects
     *
     * @param {*} value - the JS value to serialize
     * @param {number} indent - current indent level in spaces (0 for top-level value after key)
     * @param {number} depth - recursion depth (safety cap); throws if depth > 10
     * @returns {string[]} lines array (caller joins with '\n')
     */
    function serializeValue(value, indent, depth) {
      if (depth > 10) {
        throw new Error('reconstructFrontmatter: nesting depth exceeded 10 levels (pathological input)');
      }
      const pad = ' '.repeat(indent);
      const lines = [];
      if (value === null || value === undefined) {
        return ['null'];  // caller prepends key prefix; the literal 'null' means JS null on re-parse (paired with extractFrontmatter fix below)
      }
      if (Array.isArray(value)) {
        if (value.length === 0) return ['[]'];
        const allScalar = value.every(v => v === null || ['string','number','boolean'].includes(typeof v));
        const allString = value.every(v => typeof v === 'string');
        if (allString && value.length <= 3 && value.join(', ').length < 60) {
          return ['[' + value.join(', ') + ']'];  // inline
        }
        // block list
        const out = [];
        for (const item of value) {
          if (item === null || ['string','number','boolean'].includes(typeof item)) {
            const str = String(item);
            out.push(pad + '- ' + ((str.includes(':') || str.includes('#')) ? '"' + str + '"' : str));
          } else if (typeof item === 'object') {
            // Array of objects: emit `- key: value` with first key on the dash line, rest indented
            const keys = Object.keys(item);
            if (keys.length === 0) {
              out.push(pad + '- {}');
            } else {
              const [firstKey, ...restKeys] = keys;
              const firstSub = serializeValue(item[firstKey], indent + 2, depth + 1);
              if (firstSub.length === 1 && !firstSub[0].startsWith(' ')) {
                out.push(pad + '- ' + firstKey + ': ' + firstSub[0]);
              } else {
                out.push(pad + '- ' + firstKey + ':');
                for (const ln of firstSub) out.push(' '.repeat(indent + 2) + ln.trimStart() === '' ? ln : ' '.repeat(indent + 2) + ln.trimStart());
              }
              for (const k of restKeys) {
                const sub = serializeValue(item[k], indent + 2, depth + 1);
                if (sub.length === 1 && !sub[0].startsWith(' ')) {
                  out.push(' '.repeat(indent + 2) + k + ': ' + sub[0]);
                } else {
                  out.push(' '.repeat(indent + 2) + k + ':');
                  for (const ln of sub) out.push(' '.repeat(indent + 4) + ln.trimStart());
                }
              }
            }
          }
        }
        return out;
      }
      if (typeof value === 'object') {
        const keys = Object.keys(value);
        if (keys.length === 0) return ['{}'];
        const out = [];
        for (const k of keys) {
          const sub = serializeValue(value[k], indent + 2, depth + 1);
          if (sub.length === 1) {
            out.push(pad + k + ': ' + sub[0]);
          } else {
            out.push(pad + k + ':');
            for (const ln of sub) out.push(ln);
          }
        }
        return out;
      }
      // scalar
      const s = String(value);
      if (s.includes(':') || s.includes('#') || s.startsWith('[') || s.startsWith('{')) {
        return ['"' + s + '"'];
      }
      return [s];
    }
    ```
