    const fs = require('fs');
    const path = require('path');
    const { planningPaths, output } = require('./core.cjs');
    const { extractFrontmatter } = require('./frontmatter.cjs');

    function formatGate(gate) {
      if (!gate) return '— (none yet)';
      const findings = (gate.findings_count !== undefined && gate.findings_count !== null)
        ? gate.findings_count : 0;
      return `${gate.decision} (${findings} findings)`;
    }

    function getPhaseInfo(cwd) {
      // Read ROADMAP.md defensively. cmdRoadmapAnalyze may return an error object per roadmap.cjs:119
      try {
        const { cmdRoadmapAnalyze } = require('./roadmap.cjs');
        const analyzed = cmdRoadmapAnalyze(cwd, /*raw=*/true);
        // cmdRoadmapAnalyze may print via output(); prefer a direct read if available.
        // If cmdRoadmapAnalyze's return shape is unreliable, fall back to parsing ROADMAP.md directly.
        const roadmapPath = planningPaths(cwd).roadmap;
        if (!fs.existsSync(roadmapPath)) return { phase_count: '—', phase_name_short: '' };
        const content = fs.readFileSync(roadmapPath, 'utf-8');
        const phaseCount = (content.match(/^### Phase \d+:/gm) || []).length;
        return { phase_count: phaseCount || '—', phase_name_short: '' };
      } catch {
        return { phase_count: '—', phase_name_short: '' };
      }
    }

    function renderStatus(cwd, raw) {
      const statePath = planningPaths(cwd).state;
      let stateFm = {};
      let stateMissing = false;
      let briefMissing = false;

      if (!fs.existsSync(statePath)) {
        stateMissing = true;
        briefMissing = true;
      } else {
        try {
          stateFm = extractFrontmatter(fs.readFileSync(statePath, 'utf-8')) || {};
        } catch {
          stateFm = {};
        }
        briefMissing = !stateFm.brief;
      }

      const brief = (stateFm && stateFm.brief) || {};
      const currentPhase = (stateFm && stateFm.current_phase) ? stateFm.current_phase : '—';

      const { phase_count, phase_name_short } = getPhaseInfo(cwd);

      // Look up phase_name_short from ROADMAP.md (best-effort; see D-16)
      let phaseName = '';
      try {
        const roadmapPath = planningPaths(cwd).roadmap;
        if (fs.existsSync(roadmapPath)) {
          const content = fs.readFileSync(roadmapPath, 'utf-8');
          // Match `### Phase {N}: {Name}` — capture a short name
          const re = new RegExp(`^### Phase ${String(currentPhase).replace(/[^0-9.]/g, '')}:\\s*(.+?)(?:\\s*—|$)`, 'm');
          const m = content.match(re);
          if (m) phaseName = m[1].trim().split(' ').slice(0, 3).join(' ');
        }
      } catch { /* graceful */ }

      const phaseLine = `  Phase           ${currentPhase} of ${phase_count}${phaseName ? ' (' + phaseName + ')' : ''}`;
      const workstream = (brief.current_workstream) ? String(brief.current_workstream) : '— (none active)';
      const returnStackDepth = Array.isArray(brief.return_stack) ? brief.return_stack.length : 0;
      const alignLine = formatGate(brief.last_gate_results && brief.last_gate_results.align);
      const complianceLine = formatGate(brief.last_gate_results && brief.last_gate_results.compliance);
      const nextHint = (stateFm && stateFm.stopped_at) ? String(stateFm.stopped_at) : '(unknown)';

      const lines = [
        'BRIEF Status',
        '='.repeat(32),
        phaseLine,
        `  Workstream      ${workstream}`,
        `  Return stack    ${returnStackDepth} / 3`,
        `  Last ALIGN      ${alignLine}`,
        `  Last COMPLIANCE ${complianceLine}`,
        '-'.repeat(32),
        `  Next: ${nextHint}`,
      ];

      // D-17: warning if brief: map missing AND past Phase 2
      if (briefMissing) {
        // Past Phase 2 means current_phase > 02; treat numeric current_phase
        const phaseNum = parseInt(String(currentPhase).replace(/[^0-9]/g, ''), 10);
        if (stateMissing || (!Number.isNaN(phaseNum) && phaseNum >= 2)) {
          lines.push('');
          lines.push('⚠ state.brief.* not initialized — run /brief-init or check STATE.md');
        }
      }

      const rendered = lines.join('\n');
      // D-19 plain text stdout
      output({ rendered }, raw, rendered);
      return rendered;
    }
