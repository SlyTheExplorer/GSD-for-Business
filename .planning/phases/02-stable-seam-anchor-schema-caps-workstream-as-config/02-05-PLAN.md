    Public API:

    `parseYamlDocument(yamlString: string) → object` — full-doc YAML parser. NO `---` framing expected (unlike extractFrontmatter). Must handle:
    - String values (quoted and unquoted; preserve explicit quotes when stripping)
    - Number values (integer and float — coerce to JS number if parseable)
    - Boolean values (`true`, `false`, case-sensitive)
    - Null values (`null` or `~` → JS null)
    - Block lists: `key:\n  - item\n  - item` — list of strings (or list of maps; see below)
    - Inline lists: `key: [a, b, c]` — reuse `splitInlineArray`-style logic from frontmatter.cjs
    - Nested maps (multiple levels; cap at 10 per T-02-05-03)
    - List of maps (arrays of objects) — `key:\n  - subkey: val\n    subkey2: val2`

    Implementation approach: line-by-line state-machine with indent-tracking stack (adapt the algorithm from frontmatter.cjs:43-118; the KEY difference is NO `---` delimiter handling, and `parseYamlDocument` accepts the raw document). Use `Object.create(null)` for result containers. Reject `__proto__`, `constructor`, `prototype` keys (silently skip OR throw descriptive Error — pick throw for loud failure).

    Unsupported (explicitly rejected per D-12): YAML anchors (`&`, `*`), tags (`!!str`), multi-doc (`---` in middle), YAML 1.2 merge keys.

    File size target: ≤ 200 lines. If the state-machine approach exceeds 200, factor into 3 helpers (tokenize, parseBlock, coerceScalar).

    `loadWorkstreams(cwd: string) → WorkstreamSpec[]`. Implementation (adapted from 02-PATTERNS.md lines 277-327):

    ```javascript
    const fs = require('fs');
    const path = require('path');
    const { parseYamlDocument } = require('./yaml-mini.cjs');

    function loadWorkstreams(cwd) {
      const root = path.join(cwd, 'brief', 'workstreams');
      if (!fs.existsSync(root)) return [];

      // D-11: glob brief/workstreams/*/spec.yaml — no registry file
      const dirs = fs.readdirSync(root, { withFileTypes: true })
        .filter(e => e.isDirectory() && !e.name.startsWith('.'))  // T-02-05-02 dotfile exclusion (Pitfall 3)
        .map(e => e.name);

      const specs = [];
      for (const dir of dirs) {
        const workstreamDir = path.join(root, dir);
        const specPath = path.join(workstreamDir, 'spec.yaml');
        if (!fs.existsSync(specPath)) continue;  // Dir without spec.yaml is skipped, not errored (permits WIP)

        const content = fs.readFileSync(specPath, 'utf-8');
        const parsed = parseYamlDocument(content);

        // D-13 required-field validation
        if (parsed.name !== dir) {
          throw new Error(
            `Workstream "${dir}": name "${parsed.name}" does not match directory name "${dir}"`
          );
        }
        if (!parsed.description || typeof parsed.description !== 'string') {
          throw new Error(`Workstream "${dir}": missing required description`);
        }
        if (!Array.isArray(parsed.research_prompts) || parsed.research_prompts.length === 0) {
          throw new Error(`Workstream "${dir}": missing required research_prompts (non-empty list)`);
        }
        if (!Array.isArray(parsed.design_prompts) || parsed.design_prompts.length === 0) {
          throw new Error(`Workstream "${dir}": missing required design_prompts (non-empty list)`);
        }
        if (!parsed.output_artifact_template) {
          throw new Error(`Workstream "${dir}": missing required output_artifact_template`);
        }

        // T-02-05-02: directory-traversal guard for output_artifact_template
        const tmplAbs = path.resolve(workstreamDir, parsed.output_artifact_template);
        if (!tmplAbs.startsWith(workstreamDir + path.sep) && tmplAbs !== workstreamDir) {
          throw new Error(
            `Workstream "${dir}": output_artifact_template "${parsed.output_artifact_template}" resolves outside workstream directory (directory traversal rejected)`
          );
        }
        if (!fs.existsSync(tmplAbs)) {
          throw new Error(
            `Workstream "${dir}": output_artifact_template "${parsed.output_artifact_template}" does not exist`
          );
        }

        // Optional path fields — same validation
        for (const [variant, p] of Object.entries(parsed.business_model_variants || {})) {
          if (typeof p !== 'string') continue;
          const abs = path.resolve(workstreamDir, p);
          if (!abs.startsWith(workstreamDir + path.sep)) {
            throw new Error(`Workstream "${dir}": business_model_variants.${variant} "${p}" resolves outside workstream directory`);
          }
          if (!fs.existsSync(abs)) {
            throw new Error(`Workstream "${dir}": business_model_variants.${variant} path "${p}" does not exist`);
          }
        }
        for (const [region, p] of Object.entries(parsed.region_overrides || {})) {
          if (typeof p !== 'string') continue;
          const abs = path.resolve(workstreamDir, p);
          if (!abs.startsWith(workstreamDir + path.sep)) {
            throw new Error(`Workstream "${dir}": region_overrides.${region} "${p}" resolves outside workstream directory`);
          }
          if (!fs.existsSync(abs)) {
            throw new Error(`Workstream "${dir}": region_overrides.${region} path "${p}" does not exist`);
          }
        }

        specs.push({ slug: dir, ...parsed });
      }

      return specs;
    }

    module.exports = { loadWorkstreams };
    ```
